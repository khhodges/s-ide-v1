import uuid

class CapabilityToken:
    def __init__(self, resource_name, permissions):
        self.id = uuid.uuid4().hex[:8]
        self.resource_name = resource_name
        self.permissions = permissions

    def __repr__(self):
        return f"[Token:{self.id} | {self.resource_name} -> {self.permissions}]"

class ContextRegister:
    def __init__(self, name):
        self.name = name
        self.capabilities = {}

    def insert_token(self, index, token):
        self.capabilities[index] = token

    def get_token(self, index):
        return self.capabilities.get(index, None)

class ChurchMachineSimulator:
    def __init__(self):
        self.thread_stack = []
        self.current_context = ContextRegister("System-Root")
        self.active_execution_token = None
        self.threads = {}
        self.current_thread_id = "Thread-0"
        
    def LOAD(self, key_idx):
        token = self.current_context.get_token(key_idx)
        if not token:
            raise RuntimeError(f"LOAD FAULT: No capability exists at index {key_idx}!")
        self.active_execution_token = token

    def SAVE(self, token, key_idx):
        if not isinstance(token, CapabilityToken):
            raise TypeError("SAVE FAULT: Cannot forge token from raw data types!")
        self.current_context.insert_token(key_idx, token)

    def CALL(self, target_context):
        self.thread_stack.append(self.current_context)
        self.current_context = target_context

    def RETURN(self):
        if not self.thread_stack:
            raise IndexError("RETURN FAULT: Thread stack underflow!")
        self.current_context = self.thread_stack.pop()

    def CHANGE(self, target_thread_id, target_thread_context):
        self.threads[self.current_thread_id] = {
            "context": self.current_context,
            "stack": list(self.thread_stack),
            "active_token": self.active_execution_token
        }
        self.current_thread_id = target_thread_id
        if target_thread_id in self.threads:
            state = self.threads[target_thread_id]
            self.current_context = state["context"]
            self.thread_stack = state["stack"]
            self.active_execution_token = state["active_token"]
        else:
            self.current_context = target_thread_context
            self.thread_stack = []
            self.active_execution_token = None

    def SWITCH(self, new_namespace_context):
        if self.active_execution_token and "SWITCH" in self.active_execution_token.permissions:
            self.current_context = new_namespace_context
        else:
            raise PermissionError("SWITCH FAULT: Active token lacks namespace modification rights!")
