from __future__ import annotations

from common_device.rule import Rule, Checker
from design.db_item import PeriDesignItem


class BaseRuleResource(Rule):
    def __init__(self):
        super().__init__()
        self.blk_display_name = ''

    def check(self, checker: Checker, design_block: PeriDesignItem):
        device = design_block.get_device()
        if self.is_name_empty(device):
            self.error("Resource name is empty.")
            return

        assert isinstance(device, str), f'Error in self.is_name_empty()'

        # Check if resource name is a valid device instance name
        if checker.reg is not None and checker.design.device_db is not None:
            if device not in self.get_usable_resources(checker):
                self.error(f"Resource is not a valid {self.blk_display_name} device instance.")
                return

        else:
            checker.logger.warning(
                "{}: device db is empty, skip valid instance check.".format(self.name))

    def get_usable_resources(self, checker: Checker):
        return []


class BaseRuleResourceExcluded(Rule):
    def check(self, checker: Checker, design_block: PeriDesignItem):
        resource = design_block.get_device()
        if not isinstance(resource, str) or resource == "":
            return

        if self.is_resource_excluded(checker, resource):
            self.error(f"Resource {resource} is excluded in Package Planner. "\
                "Please use another resource")

    def is_resource_excluded(self, checker: Checker, resource: str):
        design = checker.design
        editor = design.get_package_editor()
        if editor is None:
            return False

        return editor.is_resource_excluded(resource)
