import re
txt = open('church_soc_cm.peri.xml').read()

# Clear CLKMUX_L ROUTE0 name — GPIO clock_name="clk" already provides the design connection
clkmux_l = re.search(r'(<efxpt:clkmux name="CLKMUX_L".*?</efxpt:clkmux>)', txt, re.DOTALL)
if clkmux_l:
    old = clkmux_l.group(1)
    new = re.sub(r'name="clk" type_name="ROUTE0"', 'name="" type_name="ROUTE0"', old, count=1)
    txt = txt.replace(old, new)
    print('OK: CLKMUX_L ROUTE0 cleared to empty')
else:
    print('WARN: CLKMUX_L block not found')

open('church_soc_cm.peri.xml', 'w').write(txt)

# Verify: one "clk" source, no duplicate
clk_names = re.findall(r'name="clk"', txt)
print(f'Total name="clk" occurrences: {len(clk_names)}')
# Should be exactly 1 (from GPIO input_config name="clk")
# clock_name="clk" and clkmux_buf_name uses different attr
for check in ['clock_name="clk"', 'clkmux_buf_name="CLKMUX_L"', 'GPIOL_P_18']:
    print(check, '->', 'FOUND' if check in txt else 'MISSING')
EOF
