#include <stdint.h>

#define UART_BASE    0xF8010000UL
#define UART_DATA    (*(volatile uint32_t*)(UART_BASE + 0x00))
#define UART_STATUS  (*(volatile uint32_t*)(UART_BASE + 0x04))
#define UART_CLOCKDIV (*(volatile uint32_t*)(UART_BASE + 0x08))

static void uart_putc(char c) {
    /* Wait for TX-available: bit 8 of STATUS */
    while (!((UART_STATUS >> 8) & 1));
    UART_DATA = (uint32_t)c;
}

static void uart_puts(const char *s) {
    while (*s) uart_putc(*s++);
}

static void delay(volatile uint32_t n) {
    while (n--);
}

int main(void) {
    /* 25 MHz / (8 * (26+1)) = 115,740 baud ~ 115200 */
    UART_CLOCKDIV = 26;
    delay(50000);   /* ~16 ms settle */

    while (1) {
        uart_puts("CHURCH Ti60 SoC+CM v1.1\r\n");
        delay(6250000);  /* ~2 s @ 25 MHz */
    }
    return 0;
}
EOF
