sipantichijk@penguin:~/church_project/SoC$ # Step 1: Find OpenOCD binary
ls ~/efinity/efinity-riscv-ide-2025.2/openocd/bin/
which openocd 2>/dev/null

# Step 2: Convert firmware ELF to raw binary
riscv64-unknown-elf-objcopy -O binary \
  ~/church_project/SoC/firmware/firmware.elf \
  ~/church_project/SoC/firmware/firmware.bin
ls -la ~/church_project/SoC/firmware/firmware.bin

# Step 3: Write the OpenOCD session script
SOC=~/efinity/2026.1/ipm/ip/efx_soc/efx_soc/fpga/fpga_ti
cat > /tmp/load_fw.cfg << 'EOF'
adapter driver ftdi
ftdi device_desc "Titanium Ti60F225 Development Kit"
ftdi vid_pid 0x0403 0x6011
ftdi channel 1
ftdi layout_init 0x08 0x0b

transport select jtag
adapter speed 1000

set _CHIPNAME riscv
jtag newtap $_CHIPNAME cpu -irlen 4 -expected-id 0x10660A79
target create $_CHIPNAME.cpu riscv -chain-position $_CHIPNAME.cpu

init
halt
echo "=== CPU HALTED, PC:"
reg pc
load_image /home/sipantichijk/church_project/SoC/firmware/firmware.bin 0xF9000000 bin
echo "=== Firmware loaded. Resuming..."
resume 0xF9000000
EOF

# Step 4: Run it (re-flash synth9 first if needed, soc.hex is fine too)
~/efinity/efinity-riscv-ide-2025.2/openocd/bin/openocd -f /tmp/load_fw.cfg
openocd
