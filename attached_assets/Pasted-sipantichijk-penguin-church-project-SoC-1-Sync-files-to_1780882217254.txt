sipantichijk@penguin:~/church_project/SoC$ # 1. Sync files to Efinity project
cp hardware/soc_combined/top.v              ~/church_project/SoC/
cp hardware/soc_combined/church_soc_cm.peri.xml ~/church_project/SoC/
cp hardware/soc_combined/firmware/main.c    ~/church_project/SoC/firmware/

# 2. Build firmware
cd ~/church_project/SoC && make -C firmware

# 3. Patch sapphire.v (MUST run every time firmware changes)
cd ~/church-machine
python3 scripts/patch_sapphire_init.py \
  ~/church_project/SoC/sapphire.v \
  ~/church_project/SoC/EfxSapphireSoc.v_toplevel_system_ramA_logic_ram_symbol0.bin \
  ~/church_project/SoC/EfxSapphireSoc.v_toplevel_system_ramA_logic_ram_symbol1.bin \
  ~/church_project/SoC/EfxSapphireSoc.v_toplevel_system_ramA_logic_ram_symbol2.bin \
  ~/church_project/SoC/EfxSapphireSoc.v_toplevel_system_ramA_logic_ram_symbol3.bin

# 4. Synthesise (~4 min)
bash hardware/soc_combined/run_efx_map.sh 2>&1 | tail -10

# 5. Verify all 4 BRAM lanes non-zero
for sym in 0 1 2 3; do
  LN=$(grep -n "EFX_RAM10" ~/church_project/SoC/outflow/church_soc_cm.map.v \
       | grep "ram_symbol${sym}__D\$g1" | head -1 | cut -d: -f1)
  echo "sym${sym}: $(sed -n "${LN},$((LN+3))p" \
       ~/church_project/SoC/outflow/church_soc_cm.map.v | grep INIT_0)"
done

# 6. Place & Route (~5 min)
bash hardware/soc_combined/run_efx_pnr.sh 2>&1 | tail -5

# 7. Generate bitstream (~30 s)
bash hardware/soc_combined/run_efx_pgm.sh 2>&1 | tail -5

# 8. Flash
sudo ~/oss-cad-suite/bin/openFPGALoader \
  -b titanium_ti60_f225_jtag \
  -f ~/church_project/SoC/outflow/church_soc_cm.hex
