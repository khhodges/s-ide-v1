sipantichijk@penguin:~/church_project/SoC$ # Tell the shell where your church-machine clone lives
# (find it with: find ~/ -name "top.v" -path "*/soc_combined*" 2>/dev/null)
CHURCH_REPO=~/church-machine          # ← change if different

# 1. Pull latest changes
cd "$CHURCH_REPO" && git pull

# 2. Copy the three updated files into the Efinity project
cp "$CHURCH_REPO/hardware/soc_combined/top.v"                ~/church_project/SoC/
cp "$CHURCH_REPO/hardware/soc_combined/church_soc_cm.peri.xml" ~/church_project/SoC/
cp "$CHURCH_REPO/hardware/soc_combined/firmware/main.c"       ~/church_project/SoC/firmware/

# Confirm the copies landed
echo "=== top.v ===" && head -5 ~/church_project/SoC/top.v
echo "=== main.c ===" && grep "FW_MINOR" ~/church_project/SoC/firmware/main.c

# 3. Build firmware
cd ~/church_project/SoC && make -C firmware

# 4. Patch sapphire.v
cd "$CHURCH_REPO"
python3 scripts/patch_sapphire_init.py \
  ~/church_project/SoC/sapphire.v \
  ~/church_project/SoC/EfxSapphireSoc.v_toplevel_system_ramA_logic_ram_symbol0.bin \
  ~/church_project/SoC/EfxSapphireSoc.v_toplevel_system_ramA_logic_ram_symbol1.bin \
  ~/church_project/SoC/EfxSapphireSoc.v_toplevel_system_ramA_logic_ram_symbol2.bin \
  ~/church_project/SoC/EfxSapphireSoc.v_toplevel_system_ramA_logic_ram_symbol3.bin

# 5. Synthesise  (uses EFX_MAP — run the script from the repo or specify paths)
bash "$CHURCH_REPO/hardware/soc_combined/run_efx_map.sh" 2>&1 | tail -10

# 6. BRAM sanity check (4 non-empty INIT_0 lines)
for sym in 0 1 2 3; do
  LN=$(grep -n "EFX_RAM10" ~/church_project/SoC/outflow/church_soc_cm.map.v \
       | grep "ram_symbol${sym}__D\$g1" | head -1 | cut -d: -f1)
  echo "sym${sym}: $(sed -n "${LN},$((LN+3))p" \
       ~/church_project/SoC/outflow/church_soc_cm.map.v | grep INIT_0)"
done

  -f ~/church_project/SoC/outflow/church_soc_cm.hexPenguin)&1 | tail -5
